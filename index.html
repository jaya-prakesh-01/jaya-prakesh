function toEmbed(url) { // Google Drive: Extract File ID let m = url.match(/\/file\/d\/([^\/\?]+)/); if(m) return { type: 'drive', src: 'https://drive.google.com/uc?export=download&id=' + m[1] }; // YouTube full m = url.match(/[?&]v=([A-Za-z0-9_\-]{11})/); if(m) return { type: 'iframe', src: 'https://www.youtube.com/embed/' + m[1] + '?autoplay=1&rel=0' }; // YouTube short m = url.match(/youtu\.be\/([A-Za-z0-9_\-]{11})/); if(m) return { type: 'iframe', src: 'https://www.youtube.com/embed/' + m[1] + '?autoplay=1&rel=0' }; // Vimeo m = url.match(/vimeo\.com\/(?:video\/)?(\d+)/); if(m) return { type: 'iframe', src: 'https://player.vimeo.com/video/' + m[1] + '?autoplay=1' }; if(url.includes('embed') || url.includes('player.vimeo') || url.includes('/preview')) { return { type: 'iframe', src: url }; } return null; } function loadSlot(n, url) { const shell = document.getElementById('vs' + n); if (!shell) return; const res = toEmbed(url); if (!res) { alert('Could not parse that URL. Try a standard Google Drive, YouTube, or Vimeo link.'); return; } shell.innerHTML = ''; shell.style.position = 'relative'; if (res.type === 'drive') { // Inject native video tag for raw drive streams const video = document.createElement('video'); video.src = res.src; video.controls = true; video.autoplay = true; video.playsInline = true; video.style.cssText = 'position:absolute;inset:0;width:100%;height:100%;object-fit:cover;background:#000;'; shell.appendChild(video); } else { // Inject standard iframe for YouTube/Vimeo const iframe = document.createElement('iframe'); iframe.src = res.src; iframe.allow = 'autoplay; fullscreen; picture-in-picture'; iframe.allowFullscreen = true; iframe.style.cssText = 'position:absolute;inset:0;width:100%;height:100%;border:none;'; shell.appendChild(iframe); } shell.classList.add('has-iframe'); if (nextSlot === n) nextSlot = n < slotCount ? n + 1 : 1; }
